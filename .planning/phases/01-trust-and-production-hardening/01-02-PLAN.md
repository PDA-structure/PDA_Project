**Case 1: UDL Simply-Supported Beam (TRUST-03, per D-06)**
```python
def test_udl_simply_supported_deflection():
    """TRUST-03: UDL simply-supported beam, analytical verification against 5wL^4/384EI."""
    w = 10000.0  # N/m
    L_beam = 4.0  # m
    # ENForces: equivalent nodal forces for UDL (positive = downward convention)
    # ENForces = [-wL/2, -wL/2] per CLAUDE.md UDL sign convention
    enf = -w * L_beam / 2
    # ENMoments = [wL^2/12, -wL^2/12]
    enm_i = w * L_beam**2 / 12
    enm_j = -w * L_beam**2 / 12

    s = BeamBarStructure_v2(
        nodes=[[0.0, 0.0], [L_beam, 0.0]],
        members=[[1, 2]],
        ENForces=[[enf, enf]],
        ENMoments=[[enm_i, enm_j]],
        force_vector=[0.0] * 6,
        E=E, I=I, A=A,
        restrainedDoF=[1, 2, 5],  # Pin at node1 (Ux1,Uy1), rollerY at node2 (Uy2)
    )
    s.solve_structure()

    # Analytical midspan deflection for simply-supported UDL: 5wL^4 / (384EI)
    expected_mid_deflection = 5 * w * L_beam**4 / (384 * E * I)
    # The solver gives nodal deflections, not midspan. But for a 2-node element,
    # we verify reactions and equilibrium. For deflection, use the rotation to
    # cross-check: theta_end for SS beam under UDL = wL^3/(24EI)
    expected_theta1 = w * L_beam**3 / (24 * E * I)

    # Reactions: each support carries wL/2
    total_applied = w * L_beam  # total downward load
    assert s.FG[1, 0] == pytest.approx(total_applied / 2, rel=1e-6)  # Uy1 reaction
    assert s.FG[4, 0] == pytest.approx(total_applied / 2, rel=1e-6)  # Uy2 reaction

    # End rotation check
    assert abs(s.UG[2, 0]) == pytest.approx(expected_theta1, rel=1e-4)

    # Equilibrium: sum(FG) + total applied = 0
    # Applied load is downward (negative in global Y), reactions are upward (positive)
    # FG contains reaction forces at restrained DOFs, zeros elsewhere
    # Total vertical reaction = FG[1] + FG[4], total applied downward = -wL (via ENForces)
    assert np.isclose(s.FG[1, 0] + s.FG[4, 0] - total_applied, 0, atol=1e-6)
```
