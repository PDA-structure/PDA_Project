4. Add `drawBMD()` function (per D-10, D-11, D-13, UI-SPEC):
```javascript
function drawBMD() {
  const moments = results.member_moments;
  if (!moments) return;

  // Auto-scale: max ordinate = 20% of shortest member pixel length (per D-12)
  let minMbrLen = Infinity;
  let maxMoment = 0;
  members.forEach((m, idx) => {
    const n1 = nodes.find(n => n.id === m.start);
    const n2 = nodes.find(n => n.id === m.end);
    if (!n1 || !n2) return;
    const dx = n2.x - n1.x, dy = n2.y - n1.y;
    const len = Math.sqrt(dx * dx + dy * dy);
    if (len < minMbrLen) minMbrLen = len;
    maxMoment = Math.max(maxMoment, Math.abs(moments[idx][0]), Math.abs(moments[idx][1]));
  });
  if (maxMoment < 1e-10) return;  // no moments to draw
  const scaleFactor = (0.2 * minMbrLen) / maxMoment;

  ctx.fillStyle = 'rgba(33, 150, 243, 0.25)';
  ctx.strokeStyle = '#1565c0';
  ctx.lineWidth = 1.5;

  members.forEach((m, idx) => {
    const n1 = nodes.find(n => n.id === m.start);
    const n2 = nodes.find(n => n.id === m.end);
    if (!n1 || !n2) return;

    const dx = n2.x - n1.x, dy = n2.y - n1.y;
    const angle = Math.atan2(dy, dx);
    // Perpendicular direction: (-sin, cos) — positive = "above" in screen coords
    const perpX = -Math.sin(angle);
    const perpY = Math.cos(angle);

    const Mi = moments[idx][0];  // moment at start node (N*m)
    const Mj = moments[idx][1];  // moment at end node (N*m)

    // BMD convention (per D-13): positive moment (sagging) draws on tension face
    // Tension face is below for horizontal members — perpendicular offset in POSITIVE direction
    // (since screen Y is inverted, positive perpY goes downward for horizontal members)
    const oi = Mi * scaleFactor;  // offset at start
    const oj = Mj * scaleFactor;  // offset at end

    ctx.beginPath();
    // Draw baseline along member
    ctx.moveTo(n1.x, n1.y);
    ctx.lineTo(n2.x, n2.y);
    // Draw offset polygon
    ctx.lineTo(n2.x + perpX * oj, n2.y + perpY * oj);

    // Check if member has UDL — if so, add midspan point for parabolic shape
    if (m.udl && m.type !== 'bar') {
      const midX = (n1.x + n2.x) / 2;
      const midY = (n1.y + n2.y) / 2;
      const L = memberLengthReal(m);
      const Mmid = (Mi + Mj) / 2 + m.udl * L * L / 8;
      const omid = Mmid * scaleFactor;
      ctx.lineTo(midX + perpX * omid, midY + perpY * omid);
    }

    ctx.lineTo(n1.x + perpX * oi, n1.y + perpY * oi);
    ctx.closePath();
    ctx.fill();
    ctx.stroke();

    // Zero-line along member axis
    ctx.strokeStyle = '#999';
    ctx.lineWidth = 1;
    ctx.beginPath();
    ctx.moveTo(n1.x, n1.y);
    ctx.lineTo(n2.x, n2.y);
    ctx.stroke();

    // Reset styles for next member
    ctx.fillStyle = 'rgba(33, 150, 243, 0.25)';
    ctx.strokeStyle = '#1565c0';
    ctx.lineWidth = 1.5;
  });
}
```

5. Add `drawSFD()` function (same pattern, different data and colors, per D-13 SFD convention):
```javascript
function drawSFD() {
  const shears = results.member_shears;
  if (!shears) return;

  let minMbrLen = Infinity;
  let maxShear = 0;
  members.forEach((m, idx) => {
    const n1 = nodes.find(n => n.id === m.start);
    const n2 = nodes.find(n => n.id === m.end);
    if (!n1 || !n2) return;
    const dx = n2.x - n1.x, dy = n2.y - n1.y;
    const len = Math.sqrt(dx * dx + dy * dy);
    if (len < minMbrLen) minMbrLen = len;
    maxShear = Math.max(maxShear, Math.abs(shears[idx][0]), Math.abs(shears[idx][1]));
  });
  if (maxShear < 1e-10) return;
  const scaleFactor = (0.2 * minMbrLen) / maxShear;

  ctx.fillStyle = 'rgba(76, 175, 80, 0.25)';
  ctx.strokeStyle = '#2e7d32';
  ctx.lineWidth = 1.5;

  members.forEach((m, idx) => {
    const n1 = nodes.find(n => n.id === m.start);
    const n2 = nodes.find(n => n.id === m.end);
    if (!n1 || !n2) return;

    const dx = n2.x - n1.x, dy = n2.y - n1.y;
    const angle = Math.atan2(dy, dx);
    const perpX = -Math.sin(angle);
    const perpY = Math.cos(angle);

    const Vi = shears[idx][0];
    const Vj = shears[idx][1];

    // SFD convention (per D-13): positive shear draws on right face (looking from i to j)
    const oi = Vi * scaleFactor;
    const oj = Vj * scaleFactor;

    ctx.beginPath();
    ctx.moveTo(n1.x, n1.y);
    ctx.lineTo(n2.x, n2.y);
    ctx.lineTo(n2.x + perpX * oj, n2.y + perpY * oj);
    ctx.lineTo(n1.x + perpX * oi, n1.y + perpY * oi);
    ctx.closePath();
    ctx.fill();
    ctx.stroke();

    // Zero-line
    ctx.strokeStyle = '#999';
    ctx.lineWidth = 1;
    ctx.beginPath();
    ctx.moveTo(n1.x, n1.y);
    ctx.lineTo(n2.x, n2.y);
    ctx.stroke();

    ctx.fillStyle = 'rgba(76, 175, 80, 0.25)';
    ctx.strokeStyle = '#2e7d32';
    ctx.lineWidth = 1.5;
  });
}
```
