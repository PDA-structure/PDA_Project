**LabelManager API:**

```js
class LabelManager {
  constructor(config) {
    // config: { ctx, members (array of {x1,y1,x2,y2} line segments for obstacle avoidance) }
    this._labels = [];
    this._placed = [];  // resolved label bounds
    this._ctx = config.ctx;
    this._memberLines = config.members || [];
  }

  add(spec) {
    // spec fields:
    //   text: string — the label text
    //   anchorX, anchorY: number — world-space anchor point (where the label "belongs")
    //   preferredX, preferredY: number|null — preferred position hint (if caller has an existing offset)
    //   priority: number — lower = placed first (10=nodeID, 20=reaction, 30=load, 40=memberForce, 50=diagram, 60=udl, 70=spring)
    //   color: string — fill colour
    //   font: string — full CSS font string (e.g. 'bold 11px Arial')
    //   fontSize: number — numeric font size in px (for bounding box computation)
    //   haloColor: string|null — stroke halo colour (null = use white background rect instead)
    //   haloWidth: number — stroke halo width (default 3)
    //   bgColor: string|null — background rect colour (null = no background)
    //   bgPadding: number — background rect padding (default 2)
    //   rotation: number|null — rotation angle in radians (for member force labels that rotate along the member)
    //   textAlign: string — 'left'|'center'|'right' (default 'center')
    //   textBaseline: string — 'top'|'middle'|'bottom' (default 'middle')
    //   radius: number — candidate position search radius (default 16)
    //   type: string — label type for debugging ('nodeId', 'dof', 'memberForce', 'load', 'reaction', 'diagram', 'udl', 'spring')
    this._labels.push(spec);
  }

  resolve() {
    // Sort by priority (ascending — lower priority number = placed first)
    this._labels.sort((a, b) => a.priority - b.priority);
    this._placed = [];

    for (const label of this._labels) {
      const placed = this._placeLabel(label);
      this._placed.push(placed);
    }
  }

  render(ctx) {
    // Draw all resolved labels. If a label was moved from its anchor, draw a leader line.
    for (const p of this._placed) {
      ctx.save();

      // Leader line (thin grey line from anchor to label center) if label moved beyond threshold
      if (p.leaderLine) {
        ctx.beginPath();
        ctx.moveTo(p.anchorX, p.anchorY);
        ctx.lineTo(p.x + p.width / 2, p.y + p.height / 2);
        ctx.strokeStyle = 'rgba(150, 150, 150, 0.6)';
        ctx.lineWidth = 0.8;
        ctx.stroke();
      }

      if (p.rotation != null) {
        ctx.translate(p.renderX, p.renderY);
        ctx.rotate(p.rotation);
        // Draw at origin after rotation
        this._drawLabelContent(ctx, p, 0, 0);
      } else {
        this._drawLabelContent(ctx, p, p.renderX, p.renderY);
      }
      ctx.restore();
    }
  }

  _placeLabel(label) {
    // Measure text width using ctx
    this._ctx.font = label.font;
    const tw = this._ctx.measureText(label.text).width;
    const th = label.fontSize * 1.3;
    const pad = label.bgPadding || 2;

    const width = tw + 2 * pad;
    const height = th + pad;

    // 1. Try preferred position if provided
    // 2. Try 8 candidate positions at radius R around anchor
    // 3. Shrink font to 80%, retry 8 candidates
    // 4. Shrink font to 65%, retry 8 candidates
    // 5. Place at extended radius (2.5x R) with leader line

    const R = label.radius || 16;
    const candidates = this._getCandidates(label.anchorX, label.anchorY, R);

    // Try preferred position first
    if (label.preferredX != null && label.preferredY != null) {
      const box = this._makeBox(label.preferredX, label.preferredY, width, height, label);
      if (!this._collides(box)) {
        return this._finalize(label, box, width, height, 1.0, false);
      }
    }

    // Try candidates at full size
    for (const c of candidates) {
      const box = this._makeBox(c.x, c.y, width, height, label);
      if (!this._collides(box)) {
        return this._finalize(label, box, width, height, 1.0, false);
      }
    }

    // Shrink to 80%
    const w80 = width * 0.8, h80 = height * 0.8;
    for (const c of candidates) {
      const box = this._makeBox(c.x, c.y, w80, h80, label);
      if (!this._collides(box)) {
        return this._finalize(label, box, w80, h80, 0.8, false);
      }
    }

    // Shrink to 65%
    const w65 = width * 0.65, h65 = height * 0.65;
    for (const c of candidates) {
      const box = this._makeBox(c.x, c.y, w65, h65, label);
      if (!this._collides(box)) {
        return this._finalize(label, box, w65, h65, 0.65, false);
      }
    }

    // Extended radius with leader line
    const extCandidates = this._getCandidates(label.anchorX, label.anchorY, R * 2.5);
    for (const c of extCandidates) {
      const box = this._makeBox(c.x, c.y, width, height, label);
      if (!this._collides(box)) {
        return this._finalize(label, box, width, height, 1.0, true);
      }
    }

    // Last resort: place at preferred or first candidate, accepting overlap
    const fallbackX = label.preferredX != null ? label.preferredX : candidates[0].x;
    const fallbackY = label.preferredY != null ? label.preferredY : candidates[0].y;
    const box = this._makeBox(fallbackX, fallbackY, width, height, label);
    return this._finalize(label, box, width, height, 1.0, false);
  }

  _getCandidates(cx, cy, R) {
    // 8 compass positions
    const cos45 = 0.707;
    return [
      { x: cx,            y: cy - R },           // top
      { x: cx + R*cos45,  y: cy - R*cos45 },     // top-right
      { x: cx + R,        y: cy },                // right
      { x: cx + R*cos45,  y: cy + R*cos45 },     // bottom-right
      { x: cx,            y: cy + R },            // bottom
      { x: cx - R*cos45,  y: cy + R*cos45 },     // bottom-left
      { x: cx - R,        y: cy },                // left
      { x: cx - R*cos45,  y: cy - R*cos45 },     // top-left
    ];
  }

  _makeBox(x, y, w, h, label) {
    // Compute bounding box based on textAlign and textBaseline
    let left, top;
    const align = label.textAlign || 'center';
    const baseline = label.textBaseline || 'middle';

    if (align === 'center') left = x - w / 2;
    else if (align === 'right') left = x - w;
    else left = x;

    if (baseline === 'middle') top = y - h / 2;
    else if (baseline === 'bottom') top = y - h;
    else top = y;

    return { left, top, right: left + w, bottom: top + h, renderX: x, renderY: y };
  }

  _collides(box) {
    // Check against all previously placed labels
    for (const p of this._placed) {
      if (!(box.right < p.left || box.left > p.right || box.bottom < p.top || box.top > p.bottom)) {
        return true;
      }
    }
    // Check against member line segments (point-to-segment distance for box center)
    const cx = (box.left + box.right) / 2;
    const cy = (box.top + box.bottom) / 2;
    for (const seg of this._memberLines) {
      if (this._pointToSegDist(cx, cy, seg.x1, seg.y1, seg.x2, seg.y2) < Math.max((box.right - box.left), (box.bottom - box.top)) / 2) {
        return true;
      }
    }
    return false;
  }

  _pointToSegDist(px, py, x1, y1, x2, y2) {
    const dx = x2 - x1, dy = y2 - y1;
    const len2 = dx * dx + dy * dy;
    if (len2 === 0) return Math.hypot(px - x1, py - y1);
    const t = Math.max(0, Math.min(1, ((px - x1) * dx + (py - y1) * dy) / len2));
    return Math.hypot(px - (x1 + t * dx), py - (y1 + t * dy));
  }

  _finalize(label, box, width, height, fontScale, leaderLine) {
    return {
      text: label.text,
      x: box.left, y: box.top,
      renderX: box.renderX, renderY: box.renderY,
      width, height,
      left: box.left, top: box.top, right: box.right, bottom: box.bottom,
      color: label.color,
      font: label.font,
      fontSize: label.fontSize,
      fontScale,
      haloColor: label.haloColor,
      haloWidth: label.haloWidth || 3,
      bgColor: label.bgColor,
      bgPadding: label.bgPadding || 2,
      rotation: label.rotation,
      textAlign: label.textAlign || 'center',
      textBaseline: label.textBaseline || 'middle',
      anchorX: label.anchorX,
      anchorY: label.anchorY,
      leaderLine,
      type: label.type,
    };
  }

  _drawLabelContent(ctx, p, x, y) {
    const scaledFontSize = Math.round(p.fontSize * p.fontScale);
    const scaledFont = p.font.replace(/\d+px/, scaledFontSize + 'px');
    ctx.font = scaledFont;
    ctx.textAlign = p.textAlign;
    ctx.textBaseline = p.textBaseline;

    // Background rect
    if (p.bgColor) {
      const pad = p.bgPadding;
      const tw = ctx.measureText(p.text).width;
      const th = scaledFontSize * 1.3;
      let bx = x, by = y;
      if (p.textAlign === 'center') bx = x - tw / 2;
      else if (p.textAlign === 'right') bx = x - tw;
      if (p.textBaseline === 'middle') by = y - th / 2;
      else if (p.textBaseline === 'bottom') by = y - th;
      ctx.fillStyle = p.bgColor;
      ctx.fillRect(bx - pad, by, tw + 2 * pad, th + pad);
    }

    // Halo stroke
    if (p.haloColor) {
      ctx.lineJoin = 'round';
      ctx.miterLimit = 2;
      ctx.lineWidth = p.haloWidth;
      ctx.strokeStyle = p.haloColor;
      ctx.strokeText(p.text, x, y);
    }

    // Fill text
    ctx.fillStyle = p.color;
    ctx.fillText(p.text, x, y);
  }
}
```
