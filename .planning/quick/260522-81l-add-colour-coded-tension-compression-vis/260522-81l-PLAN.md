   ```js
   function drawReactions() {
     if (!results || !results.FG) return;
     const sc = getSymbolScale();
     const arrowLen = 24 * sc;   // same base length as drawLoads — uniform, not magnitude-scaled
     const arrowTip = 19 * sc;
     const arrowHW  = 5 * sc;
     const fs       = Math.round(10 * sc);
     const ZERO     = 1e-3;       // match member zero-tolerance per CONTEXT Claude's Discretion
     const FG       = results.FG;

     ctx.save();
     ctx.strokeStyle = '#7b1fa2';
     ctx.fillStyle   = '#7b1fa2';
     ctx.lineWidth   = 2;
     ctx.font        = `${fs}px Arial`;
     ctx.textAlign   = 'center';

     supports.forEach(s => {
       const n = nodes.find(nd => nd.id === s.nodeId);
       if (!n) return;
       const base = s.nodeId * 2;
       // For each restrained direction, draw arrow pointing in direction of +reaction
       const dirs = s.type === 'pinned'  ? ['x', 'y']
                  : s.type === 'rollerX' ? ['x']
                  : s.type === 'rollerY' ? ['y']
                  : [];
       dirs.forEach(dir => {
         const idx = base + (dir === 'y' ? 1 : 0);
         const r   = FG[idx];
         if (Math.abs(r) &lt; ZERO) return;   // skip arrows for zero-length nothings

         // Arrow points in the direction the support pushes the structure.
         // Canvas y is flipped: positive Y reaction (upward in world) draws as -y in canvas.
         if (dir === 'y') {
           const sign = r > 0 ? -1 : 1;   // positive reaction => upward world => -y canvas
           ctx.beginPath(); ctx.moveTo(n.x, n.y); ctx.lineTo(n.x, n.y + sign * arrowLen); ctx.stroke();
           ctx.beginPath();
           ctx.moveTo(n.x - arrowHW, n.y + sign * arrowTip);
           ctx.lineTo(n.x, n.y + sign * arrowLen);
           ctx.lineTo(n.x + arrowHW, n.y + sign * arrowTip);
           ctx.fill();
           const labelY = n.y + sign * (arrowLen + 12 * sc);
           ctx.fillText((Math.abs(r) / 1000).toFixed(2) + ' kN', n.x, labelY);
         } else {
           const sign = r > 0 ? 1 : -1;   // positive reaction => +x world => +x canvas (no flip)
           ctx.beginPath(); ctx.moveTo(n.x, n.y); ctx.lineTo(n.x + sign * arrowLen, n.y); ctx.stroke();
           ctx.beginPath();
           ctx.moveTo(n.x + sign * arrowTip, n.y - arrowHW);
           ctx.lineTo(n.x + sign * arrowLen, n.y);
           ctx.lineTo(n.x + sign * arrowTip, n.y + arrowHW);
           ctx.fill();
           const labelX = n.x + sign * (arrowLen + 18 * sc);
           ctx.fillText((Math.abs(r) / 1000).toFixed(2) + ' kN', labelX, n.y + 4);
         }
       });
     });

     ctx.restore();
   }
   ```

1. Add `drawLegend()` above `drawReactions()` (or anywhere in the draw-helpers cluster). Suggested implementation — vertical list inside a padded semi-transparent white card so it reads cleanly over both the grid and any member that runs underneath it:

   ```js
   function drawLegend() {
     if (!results) return;
     const sc = getSymbolScale();

     // Legend lives in SCREEN space, not world space — reset transform so pan/zoom doesn't shift it.
     ctx.save();
     ctx.setTransform(1, 0, 0, 1, 0, 0);

     const fs       = Math.round(11 * sc);
     const lh       = Math.round(16 * sc);    // line height
     const swatchW  = Math.round(22 * sc);    // sample line width
     const padX     = Math.round(10 * sc);
     const padY     = Math.round(8 * sc);
     const gap      = Math.round(8 * sc);     // gap between swatch and text

     const items = [
       { color: '#1565c0', label: 'Tension (+)' },
       { color: '#b71c1c', label: 'Compression (-)' },
       { color: '#999',    label: 'Near-zero' },
       { color: '#7b1fa2', label: 'Reaction' },
     ];

     ctx.font = `${fs}px Arial`;
     // Compute box width from longest label
     let maxTextW = 0;
     items.forEach(it => { maxTextW = Math.max(maxTextW, ctx.measureText(it.label).width); });
     const boxW = padX * 2 + swatchW + gap + Math.ceil(maxTextW);
     const boxH = padY * 2 + items.length * lh;

     // Anchor to top-right of canvas with a small margin
     const margin = Math.round(10 * sc);
     const x0 = canvas.width  - boxW - margin;
     const y0 = margin;

     // Card background (semi-transparent white so grid + members behind stay faintly visible)
     ctx.fillStyle   = 'rgba(255, 255, 255, 0.92)';
     ctx.strokeStyle = '#ccc';
     ctx.lineWidth   = 1;
     ctx.beginPath();
     ctx.rect(x0, y0, boxW, boxH);
     ctx.fill();
     ctx.stroke();

     // Rows
     ctx.textAlign    = 'left';
     ctx.textBaseline = 'middle';
     items.forEach((it, i) => {
       const rowY = y0 + padY + i * lh + Math.round(lh / 2);
       // Swatch (line or short stroke matching member colour; reaction uses an arrow head for clarity)
       ctx.strokeStyle = it.color;
       ctx.fillStyle   = it.color;
       ctx.lineWidth   = 3;
       ctx.beginPath();
       ctx.moveTo(x0 + padX, rowY);
       ctx.lineTo(x0 + padX + swatchW, rowY);
       ctx.stroke();
       // Label
       ctx.fillStyle = '#222';
       ctx.fillText(it.label, x0 + padX + swatchW + gap, rowY);
     });

     ctx.restore();
   }
   ```
